create database Alunos;
use Alunos;

CREATE TABLE Aluno (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Nome VARCHAR(100),
    RA VARCHAR(30),
    CPF VARCHAR(14),
    DT_Nasc DATE
);


CREATE TABLE Aluno_Login (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Aluno_ID INT,
    Email VARCHAR(30),
    Senha VARCHAR(30),
    FOREIGN KEY (Aluno_ID) REFERENCES Aluno(ID) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE Aluno_Endereco (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Aluno_ID INT,
    Rua VARCHAR(100),
    Bairro VARCHAR(100),
    CEP VARCHAR(14),
    Pais VARCHAR(30),
    UF VARCHAR(2),
    FOREIGN KEY (Aluno_ID) REFERENCES Aluno(ID) ON DELETE CASCADE ON UPDATE CASCADE
);

CREATE TABLE Aluno_Contato(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Aluno_ID INT,
    Telefone VARCHAR(100),
    Celular VARCHAR(100),
    FOREIGN KEY (Aluno_ID) REFERENCES Aluno(ID) ON DELETE CASCADE ON UPDATE CASCADE
);


CREATE TABLE Aluno_Ano(
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Aluno_ID INT,
    Ano VARCHAR(1),
    Turma VARCHAR(1),
    FOREIGN KEY (Aluno_ID) REFERENCES Aluno(ID) ON DELETE CASCADE ON UPDATE CASCADE
);
